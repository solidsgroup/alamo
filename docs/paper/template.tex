% Uses your YAML:
% authors: [{name, orcid, affiliation: <index>}]
% affiliations: [{name, index: <index>}]
%
% We render authors with authblk, mapping each author's "affiliation" number
% to an \affil[...] entry.

$if(authors)$

% Define each author with their affiliation index as authblk "key"
$for(authors)$
\author[$authors.affiliation$]{%
  $authors.name$
  $if(authors.orcid)$%
    \,\orcidlink{$authors.orcid$}%
  $endif$%
}
$endfor$

% Define each affiliation; the "index" becomes the numeric key

$for(affiliations)$
\affil[$affiliations.index$]{$affiliations.name$}
$endfor$
